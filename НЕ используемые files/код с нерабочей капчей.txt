     <script>
        // Функция для получения капчи с сервера
        function getCaptcha() {
            // Показываем индикатор загрузки
            document.getElementById('captchaText').innerHTML = 'Loading CAPTCHA...';
            document.getElementById('captchaText').classList.add('captcha-loading');
            
            // Отправляем AJAX-запрос к серверу
            fetch('/includes/captcha.php')
                .then(response => {
                    if (!response.ok) {
                        throw new Error('Network response was not ok');
                    }
                    return response.json();
                })
                .then(data => {
                    // Отображаем капчу
                    document.getElementById('captchaText').textContent = data.expression;
                    document.getElementById('captchaText').classList.remove('captcha-loading');
                    
                    // Сохраняем ответ в data-атрибут для последующей проверки
                    document.getElementById('captchaText').setAttribute('data-answer', data.answer);
                })
                .catch(error => {
                    console.error('Error fetching CAPTCHA:', error);
                    document.getElementById('captchaText').textContent = 'Error loading CAPTCHA';
                    document.getElementById('captchaText').classList.remove('captcha-loading');
                    
                    // Через 2 секунды пытаемся снова
                    setTimeout(getCaptcha, 2000);
                });
        }
        
        // Инициализация капчи при загрузке страницы
        document.addEventListener('DOMContentLoaded', function() {
            getCaptcha();
            
            // Добавляем валидацию капчи при отправке формы
            document.getElementById('uploadForm').addEventListener('submit', function(e) {
                const userAnswer = document.getElementById('captcha').value;
                const correctAnswer = document.getElementById('captchaText').getAttribute('data-answer');
                
                if (userAnswer !== correctAnswer) {
                    e.preventDefault();
                    alert('Incorrect CAPTCHA answer. Please try again.');
                    getCaptcha();
                    document.getElementById('captcha').value = '';
                }
            });
        });
    </script>
